def read_and_extract(input_file, output_file):
    with open(input_file, 'r') as file:
        lines = file.readlines()

    extracted_content = []
    for line in lines:
        # 评估元组字符串，转换为元组对象
        tup = eval(line.strip())
        # 提取索引为1的元素
        extracted_content.append(tup[1])

    with open(output_file, 'w') as file:
        for item in extracted_content:
            file.write(str(item) + '\n')

# 使用函数读取1.txt并写入2.txt
read_and_extract('1.txt', '2.txt')
